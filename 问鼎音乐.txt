function updateProgress(){
  if(audio.duration){
    const pct=(audio.currentTime/audio.duration)*100
    document.getElementById('progressBar').style.width=pct+'%'
    document.getElementById('currentTime').textContent=formatTime(audio.currentTime)
    document.getElementById('durationTime').textContent=formatTime(audio.duration)
    isPlaying=!audio.paused; updatePlayButton(isPlaying)
    // 歌词同步
    if(currentLyric.length){
      let activeIdx=-1
      for(let i=0;i<currentLyric.length;i++){ if(audio.currentTime>=currentLyric[i].time)activeIdx=i; else break }
      if(activeIdx<0)activeIdx=0
      if(activeIdx!==lastActiveIdx){
        lastActiveIdx=activeIdx
        const line=currentLyric[activeIdx]
        document.getElementById('miniLyricText').textContent=line?.text?`🎤 ${line.text}`:'🎵 ...'
        document.querySelectorAll('.lyric-line').forEach((el,i)=>{
          el.className=i===activeIdx ? 'lyric-line block text-sm md:text-base text-brand-400 font-bold transition-all duration-300 py-1' : 'lyric-line block text-xs md:text-sm text-slate-500 transition-all duration-200 py-0.5'
          if(i===activeIdx){ const c=el.closest('#lyricsContainer'); if(c){ const top=el.offsetTop-c.clientHeight/2+el.clientHeight/2; c.scrollTo({top,behavior:'smooth'}) } }
        })
      }
    }
    // 全屏模式同步
    if(isFullscreen&&audio.duration){
      document.getElementById("fsProgressBar").style.width=(audio.currentTime/audio.duration*100)+"%";
      document.getElementById("fsCurrent").textContent=formatTime(audio.currentTime);
      updateFsLyric()
    }
  }
}

async function loadLyric(rid){
  currentLyric=[]; lastActiveIdx=-1

  try{

        document.getElementById('miniLyricText').textContent=`🎵 ${currentSong?.name||''}`
      } else { lc.innerHTML='<div class="text-center text-slate-600 py-8 text-xs">纯音乐，无歌词</div>'; document.getElementById('miniLyricText').textContent='🔇 无歌词' }

async function downloadSong(rid,name,artist){
  showToast('⏳ 获取直链...')
  try{
    const r=await fetch('/api/download',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({rid,artist,title:name,bitrate:currentBitrate})})
    const d=await r.json()
    if(d.success){
      // 方式1：用 window.open 打开CDN直链（浏览器直接下载，不经过服务器）
      const a=document.createElement('a'); a.href=d.url; a.download=d.filename||(name+'.mp3'); document.body.appendChild(a); a.click(); setTimeout(()=>document.body.removeChild(a),1000)
      showToast('✅ 正在下载: '+(d.filename||name+'.mp3'))
      loadHistory()
    } else showToast('❌ '+(d.message||'获取失败'))
  } catch(e){ showToast('❌ 网络错误') }
}

async function goDonate(){
  let amount=selectedAmount
  const custom=document.getElementById('customAmount').value
  if(custom&&parseFloat(custom)>0) amount=custom+'元'
  if(!amount){showToast('请选择或输入金额');return}
  const name=document.getElementById('donateName').value.trim()||'热心听友'
  showToast('⏳ 生成订单...')
  try{
    const r=await fetch('/api/donate/preorder',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({name,amount})})
    const d=await r.json()
    if(d.success){
      localStorage.setItem('donateOrder',JSON.stringify({order_id:d.order_id,name:d.name,amount}))
      const p=await fetch('/api/donate/payurl').then(r=>r.json())
      if(p.success) window.open(p.url,'_blank')
      showToast('✅ 订单已生成，支付后回来确认')
      // 确认按钮
      showDonateConfirm(d.order_id)
    } else showToast('❌ '+d.message)
  } catch(e){showToast('❌ 网络错误')}
}
function showDonateConfirm(orderId){
  const old=document.getElementById('donateConfirmSection'); if(old)old.remove()
  const sec=document.createElement('div'); sec.id='donateConfirmSection'; sec.className='pt-2'
  sec.innerHTML='<button class="w-full bg-gradient-to-r from-green-500 to-emerald-500 text-white font-bold py-2 rounded-xl transition active:scale-[0.98]">✅ 已支付，确认登记</button>'
  sec.querySelector('button').onclick=async()=>{
    showToast('⏳ 确认中...')
    const r=await fetch('/api/donate/confirm',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({order_id:orderId})})
    const d=await r.json()
    if(d.success){showToast('🎉 '+d.message);closeDonate();localStorage.removeItem('donateOrder');sec.remove()}
    else showToast('❌ '+d.message)
  }
  document.querySelector('#donateModal .p-4').appendChild(sec)
}
